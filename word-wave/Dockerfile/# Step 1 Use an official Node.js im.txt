# Step 1: Use an official Node.js image for building the app
FROM node:16 AS build

# Step 2: Set the working directory
WORKDIR /app

# Step 3: Copy package.json and package-lock.json
COPY package*.json ./

# Step 4: Install dependencies
RUN npm install

# Step 5: Copy the rest of the application files
COPY . .

# Step 6: Build the application
RUN npm run build

# Step 7: Use an official NGINX image to serve the built files
FROM nginx:alpine

# Step 8: Copy built files from the previous stage to NGINX's html directory
COPY --from=build /app/dist /usr/share/nginx/html

# Step 9: Copy custom NGINX configuration file (optional)
# COPY ./nginx.conf /etc/nginx/conf.d/default.conf

# Step 10: Expose port 80
EXPOSE 80

# Step 11: Start NGINX server
CMD ["nginx", "-g", "daemon off;"]
